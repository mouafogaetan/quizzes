Introduction au Langage C

Définitions importantes

  • Programmation: Traduction d'un algorithme en langage de programmation
  • Langage de programmation: Ensemble des mots et symboles permettant d'écrire un programme
  • Programme: Suite d'instructions écrite dans un langage de programmation permettant de réaliser une ou plusieurs tâches
  • Compilateur: Application qui transforme le code source en fichier binaire exécutable (ex: GCC, Visual Studio)

Structure d'un programme C

#include <stdio.h>  // Directive au préprocesseur

int main() {          // Fonction principale
    // Déclarations de variables
    int a = 5;
    
    // Instructions
    printf("Hello World! %d", a);
    
    return 0;         // Fin du programme
}

Bibliothèques standard

Bibliothèque Rôle
<stdio.h> Fonctions d'entrée/sortie (printf, scanf)
<stdlib.h> Opérations de conversion, allocation mémoire
<math.h> Fonctions mathématiques (sqrt, pow)
<string.h> Manipulation des chaînes de caractères

Exercice 1: Premier programme

Écrivez un programme qui demande à l'utilisateur son nom et son âge, puis affiche un message personnalisé.

#include <stdio.h>

int main() {
    char nom[50];
    int age;
    
    printf("Entrez votre nom: ");
    scanf("%s", nom);
    
    printf("Entrez votre âge: ");
    scanf("%d", &age);
    
    printf("Bonjour %s, vous avez %d ans.\n", nom, age);
    
    return 0;
}

Structures de Contrôle

Structure conditionnelle if-else

if (condition) {
    // Instructions si condition vraie
} else {
    // Instructions si condition fausse
}

Exemple:

int n;
printf("Saisir un nombre: ");
scanf("%d", &n);

if (n % 2 == 0) {
    printf("%d est pair\n", n);
} else {
    printf("%d est impair\n", n);
}

Opérateur conditionnel ternaire

Syntaxe: condition ? expression1 : expression2

int a = 5, b = 10;
int max = (a > b) ? a : b;  // max contiendra 10

Boucles

Boucle for (nombre d'itérations connu)

for (initialisation; condition; incrémentation) {
    // Instructions à répéter
}

Exemple:

for (int i = 0; i < 5; i++) {
    printf("i = %d\n", i);
}

Boucle while (condition d'arrêt)

while (condition) {
    // Instructions à répéter
}

Boucle do-while (au moins une exécution)

do {
    // Instructions à répéter
} while (condition);

Exercice 2: Table de multiplication

Écrivez un programme qui demande un nombre à l'utilisateur et affiche sa table de multiplication de 1 à 10.

#include <stdio.h>

int main() {
    int n;
    
    printf("Entrez un nombre: ");
    scanf("%d", &n);
    
    printf("Table de multiplication de %d:\n", n);
    for (int i = 1; i <= 10; i++) {

Les Tableaux

Déclaration et initialisation

// Déclaration d'un tableau de 10 entiers
int tab[10];

int nombres[5] = {1, 2, 3, 4, 5};

// Accès aux éléments (index commence à 0)
int premier = nombres[0];  // = 1
int troisieme = nombres[2]; // = 3

Parcours d'un tableau

#define TAILLE 5

int main() {
    int tab[TAILLE] = {10, 20, 30, 40, 50};
    
    // Parcours avec for
    for (int i = 0; i < TAILLE; i++) {
        printf("tab[%d] = %d\n", i, tab[i]);
    }
    
    return 0;
}

Les Fonctions (Sous-programmes)

Définition d'une fonction

// Fonction qui retourne une valeur
int carre(int x) {
    return x * x;
}

// Fonction qui ne retourne rien (void)
void afficherMessage() {
    printf("Bonjour!\n");
}

Appel de fonction

int main() {
    int resultat = carre(5);  // resultat = 25
    afficherMessage();        // Affiche "Bonjour!"
    
    return 0;
}

Variables locales et globales

int globale = 10;  // Variable globale

void maFonction() {

}

Exercice 4: Fonction de calcul

Écrivez une fonction qui calcule la factorielle d'un nombre entier positif, puis testez-la dans un programme principal.

#include <stdio.h>

// Déclaration de la fonction
int factorielle(int n);

int main() {
    int n;
    
    printf("Entrez un nombre entier positif: ");
    scanf("%d", &n);
    
    if (n < 0) {
        printf("Erreur: le nombre doit être positif.\n");
    } else {
        printf("%d! = %d\n", n, factorielle(n));
    }
    
    return 0;
}

// Définition de la fonction
int factorielle(int n) {
    int resultat = 1;
    
    for (int i = 1; i <= n; i++) {
        resultat *= i;
    }
    
    return resultat;
}

Exécution d'un Programme C

Processus de compilation

  1. Préprocessing: Traitement des directives (#include, #define)
  2. Compilation: Traduction du code C en langage assembleur
  3. Assemblage: Conversion de l'assembleur en code machine (fichier objet)
  4. Édition de liens: Liaison des fichiers objets et bibliothèques pour créer l'exécutable

Compilation avec GCC

# Compilation simple
gcc programme.c -o programme

# Compilation avec bibliothèque mathématique
gcc programme.c -o programme -lm

# Affichage des warnings
gcc -Wall programme.c -o programme

Exercice 5: Programme complet

Écrivez un programme complet qui:

  1. Demande à l'utilisateur de saisir 5 nombres
  2. Les stocke dans un tableau
  3. Affiche le maximum, le minimum et la moyenne en utilisant des fonctions
#include <stdio.h>

#define TAILLE 5

// Prototypes des fonctions
float trouverMax(float tab[], int taille);
float trouverMin(float tab[], int taille);
float calculerMoyenne(float tab[], int taille);

int main() {
    float nombres[TAILLE];
    
    // Saisie des nombres
    printf("Entrez %d nombres:\n", TAILLE);
    for (int i = 0; i < TAILLE; i++) {
        printf("Nombre %d: ", i+1);
        scanf("%f", &nombres[i]);
    }
    
    // Calculs et affichage
    printf("\nMaximum: %.2f\n", trouverMax(nombres, TAILLE));
    printf("Minimum: %.2f\n", trouverMin(nombres, TAILLE));
    printf("Moyenne: %.2f\n", calculerMoyenne(nombres, TAILLE));
    
    return 0;
}

// Fonction pour trouver le maximum
float trouverMax(float tab[], int taille) {
    float max = tab[0];
    for (int i = 1; i < taille; i++) {
        if (tab[i] > max) {
            max = tab[i];
        }
    }
    return max;
}

// Fonction pour trouver le minimum
float trouverMin(float tab[], int taille) {
    float min = tab[0];
    for (int i = 1; i < taille; i++) {
        if (tab[i] < min) {
            min = tab[i];
        }
    }
    return min;
}

// Fonction pour calculer la moyenne
float calculerMoyenne(float tab[], int taille) {
    float somme = 0;
    for (int i = 0; i < taille; i++) {
        somme += tab[i];
    }
    return somme / taille;
}
